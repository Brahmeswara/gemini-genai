package com.example.lws.models;

import static org.assertj.core.api.Assertions.assertThat;
import java.math.BigDecimal;
import org.junit.jupiter.api.Test;

class BookTest {

    @Test
    void testCreateBook() {
        Publisher publisher = Publisher.builder().id(1L).name("Packt").build();

        Book book = Book.builder()
                .title("Mockito in Action")
                .author("Elliotte Rusty Harold")
                .publisher(publisher)
                .yearOfPublication(2007)
                .isbn("978-1-84719-284-5")
                .price(BigDecimal.valueOf(9.99))
                .quantityInStock(10)
                .quantityAvailable(10)
                .build();

        assertThat(book).isNotNull();
        assertThat(book.getId()).isEqualTo(null);
        assertThat(book.getTitle()).isEqualTo("Mockito in Action");
        assertThat(book.getAuthor()).isEqualTo("Elliotte Rusty Harold");
        assertThat(book.getPublisher()).isEqualTo(publisher);
        assertThat(book.getYearOfPublication()).isEqualTo(2007);
        assertThat(book.getIsbn()).isEqualTo("978-1-84719-284-5");
        assertThat(book.getPrice()).isEqualTo(BigDecimal.valueOf(9.99));
        assertThat(book.getQuantityInStock()).isEqualTo(10);
        assertThat(book.getQuantityAvailable()).isEqualTo(10);
    }

    @Test
    void testUpdateBook() {
        Publisher publisher = Publisher.builder().id(1L).name("Packt").build();
        Book book = Book.builder()
                .id(1L)
                .title("Mockito in Action")
                .author("Elliotte Rusty Harold")
                .publisher(publisher)
                .yearOfPublication(2007)
                .isbn("978-1-84719-284-5")
                .price(BigDecimal.valueOf(9.99))
                .quantityInStock(10)
                .quantityAvailable(10)
                .build();

        book.setTitle("Mockito in Action Second Edition");
        book.setPrice(BigDecimal.valueOf(12.99));
        book.setQuantityInStock(15);

        assertThat(book).isNotNull();
        assertThat(book.getId()).isEqualTo(1L);
        assertThat(book.getTitle()).isEqualTo("Mockito in Action Second Edition");
        assertThat(book.getAuthor()).isEqualTo("Elliotte Rusty Harold");
        assertThat(book.getPublisher()).isEqualTo(publisher);
        assertThat(book.getYearOfPublication()).isEqualTo(2007);
        assertThat(book.getIsbn()).isEqualTo("978-1-84719-284-5");
        assertThat(book.getPrice()).isEqualTo(BigDecimal.valueOf(12.99));
        assertThat(book.getQuantityInStock()).isEqualTo(15);
        assertThat(book.getQuantityAvailable()).isEqualTo(10);
    }
}
